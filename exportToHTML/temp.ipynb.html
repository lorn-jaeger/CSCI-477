11   def projectile(t,y,b):
12     x, y, vx, vy = y
13     speed = np.sqrt(vx**2 + vy**2)
14   
15     g = b.g
16     drag = b.get_drag(speed)
17   
18     u_x = vx / speed if speed != 0 else 0
19     u_y = vy / speed if speed != 0 else 0
20   
21     ax = -drag * u_x
22     ay = g - drag * u_y
23   
24     return np.array([vx, vy, ax, ay])
